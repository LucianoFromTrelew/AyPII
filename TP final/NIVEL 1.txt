Procedimiento ABM(E opc: entero, E/S clientes: arbolClientes, E/S vehiculos: arbolVehiculos, E/S modelos: listaModelos)
{}
VAR
	opc1: entero
INICIO
	Repetir
		mostrarMenuABM()
		opc1 <- enteroEnRango("Ingrese su opción", 1, 6)
		opc1 <- (opc * 10) + opc1
		Segun (opc1) hacer
			11: altaModelo(modelos)
			12: altaEtapas(modelos)
			13: altaClientes(clientes)
			14: altaVehiculos(vehiculos, clientes, modelos)
			15: altaServicios(clientes, vehiculos, modelos)
			21: bajaModelo(modelos, clientes, vehiculos)
			22: bajaEtapas(modelos, vehiculos)
			23: bajaClientes(clientes, vehiculos)
			24: bajaVehiculos(clientes, vehiculos)
			25: bajaServicios(vehiculos)
			31: modModelo(modelos)
			32: modEtapas(modelos)
			33: modClientes(clientes)
			34: modVehiculos(vehiculos, clientes, modelos)
			35: modServicios(vehiculos)
		Fin Segun
	Hasta que (opc1 = 16) o (opc1 = 26) o (opc1 = 36)
	
FIN

	
Procedimiento mostrarMenuABM()
INICIO
	mostrar("1. Modelos")
	mostrar("2. Etapas")
	mostrar("3. Clientes")
	mostrar("4. Vehículos de un cliente")
	mostrar("5. Servicios realizados sobre un vehículo")
	mostrar("6. Salir")
FIN


Procedimiento Consultas(E clientes: arbolClientes, E vehiculos, arbolVehiculos, E modelos: listaModelos)
VAR
	opc: entero
INICIO
	Repetir
		mostrarMenuConsultas()
		opc <- enteroEnRango("Ingrese su opción", 1, 4)
		Segun (opc) hacer
			1: mantXCliente(clientes, vehiculos)
			2: mantXMod()
			3: datosClientes()
		Fin Segun
	Hasta que (opc = 4)
FIN


Procedimiento mostrarMenuConsultas()
INICIO
	mostrar("1. Mantenimientos de un cliente")
	mostrar("2. Mantenimientos por modelo")
	mostrar("3. Datos de clientes sin mantenimientos realizados")
	mostrar("4. Salir")
FIN
